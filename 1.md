cd 定位到相关文件位置

mrdir 创建新目录也就是文件夹

rmdir 删除空目录

touch 创建新文件

git init 初始化仓库

git clone url 克隆远程仓库

git log 查看提交历史

git status 查看状态

git add 文件 添加文件到暂存区

git add . 添加所有文件

git commit -m “消息”

git branch 查看分支   #现在还不是很理解分支的概念，暂时没有使用场景

git branch 分支名 创建分支

git remote -v 查看远程仓库

git push origin 分支名 推送到远程

git pull origin 分支名 从远处拉取

git fetch 获取远程更新

find . -name "文件名"  查找文件
